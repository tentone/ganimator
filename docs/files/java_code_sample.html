package com.tentone.ganimator.test; import com.badlogic.gdx.ApplicationAdapter; import com.badlogic.gdx.Gdx; import com.badlogic.gdx.graphics.GL20; import com.badlogic.gdx.graphics.OrthographicCamera; import com.badlogic.gdx.graphics.g2d.SpriteBatch; public class MainTest extends ApplicationAdapter { private static SpriteBatch batch; private static OrthographicCamera camera; private static Animation animation; private static TextureManager manager; @Override public void create() { try { iniCamera(Gdx.graphics.getWidth(),Gdx.graphics.getHeight()); manager = new TextureManager("animation/"); animation = new Animation(manager); animation.loadFile("knigth_walking.gsa"); animation.setPosition(0,0); } catch (Exception e) { e.printStackTrace(); Gdx.app.exit(); } } @Override public void render() { Gdx.gl.glClearColor(0.3f, 0.3f, 0.3f, 1); Gdx.gl.glClear(GL20.GL_COLOR_BUFFER_BIT); batch.setProjectionMatrix(camera.combined); batch.begin(); animation.update(1f/60f); animation.draw(batch); batch.end(); } public static void iniCamera(float x, float y) { batch = new SpriteBatch(); camera = new OrthographicCamera(); camera.setToOrtho(false,x,y); camera.position.set(0f,50f,0f); camera.zoom=0.35f; camera.update(); } }
